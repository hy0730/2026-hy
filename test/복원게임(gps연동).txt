<script>
    // === 1. GPS 설정 (안양 만안교 좌표) ===
    const TARGET_LAT = 37.4019; // 만안교 위도 (예시)
    const TARGET_LNG = 126.9223; // 만안교 경도
    const UNLOCK_DISTANCE = 50; // 50m 이내 접근 시 해금

    const distTxt = document.getElementById('distance-txt');
    const statusTxt = document.getElementById('status-txt');
    const btnCamera = document.getElementById('btn-camera');

    // 위치 추적 시작
    if (navigator.geolocation) {
        navigator.geolocation.watchPosition(updatePosition, handleError, {
            enableHighAccuracy: true
        });
    } else {
        distTxt.innerText = "GPS 지원 불가";
    }

    function updatePosition(position) {
        const userLat = position.coords.latitude;
        const userLng = position.coords.longitude;
        
        // 거리 계산 (Haversine 공식)
        const dist = getDistanceFromLatLonInM(userLat, userLng, TARGET_LAT, TARGET_LNG);
        
        // 화면 표시
        if (dist > 1000) {
            distTxt.innerText = (dist / 1000).toFixed(1) + " km";
        } else {
            distTxt.innerText = Math.floor(dist) + " m";
        }

        // 거리별 반응
        if (dist <= UNLOCK_DISTANCE) {
            statusTxt.innerText = "목표 지점에 도착했습니다!";
            statusTxt.style.color = "#00ff88";
            btnCamera.style.display = "block"; // 버튼 등장
        } else {
            statusTxt.innerText = "목표를 향해 이동하세요.";
            btnCamera.style.display = "none";
        }
    }

    function handleError(error) {
        statusTxt.innerText = "위치 정보를 가져올 수 없습니다.";
    }

    // 거리 계산 수학 공식 (미터 단위)
    function getDistanceFromLatLonInM(lat1, lon1, lat2, lon2) {
        var R = 6371000; // 지구 반지름 (미터)
        var dLat = deg2rad(lat2 - lat1);
        var dLon = deg2rad(lon2 - lon1);
        var a = Math.sin(dLat/2) * Math.sin(dLat/2) +
                Math.cos(deg2rad(lat1)) * Math.cos(deg2rad(lat1)) *
                Math.sin(dLon/2) * Math.sin(dLon/2);
        var c = 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1-a));
        var d = R * c;
        return d;
    }
    function deg2rad(deg) { return deg * (Math.PI/180); }


    // === 2. 카메라 촬영 및 연동 ===
    const cameraInput = document.getElementById('camera-input');
    const imgEl = document.getElementById('target-img');
    
    function openCamera() {
        cameraInput.click(); // 숨겨진 파일 인풋 클릭
    }
    
    // 개발자용 강제 실행 (집에서 테스트할 때)
    function forceFound() {
        alert("테스트 모드: 강제로 카메라를 켭니다.");
        openCamera();
    }

    // 사진을 찍거나 선택했을 때
    cameraInput.addEventListener('change', function(e) {
        const file = e.target.files[0];
        if (file) {
            const reader = new FileReader();
            reader.onload = function(evt) {
                // 찍은 사진을 게임 이미지로 설정
                imgEl.src = evt.target.result;
                
                // 화면 전환 (탐색 -> 복원)
                document.getElementById('exploration-screen').style.display = 'none';
                document.getElementById('restore-screen').style.display = 'flex';
                
                alert("📷 유물 확보 완료!\n훼손된 유물을 복원해주세요!");
            };
            reader.readAsDataURL(file);
        }
    });


    // === 3. 복원 게임 로직 (아까 그 코드 연동) ===
    let restoration = 0;
    const hpBar = document.getElementById('hp-bar');
    const percentTxt = document.getElementById('percent-txt');
    const msgTxt = document.getElementById('msg-txt');

    function restoreAction(amount) {
        if (restoration >= 100) return;

        restoration += amount;
        if (restoration > 100) restoration = 100;

        // 시각 효과: 흑백/흐림 -> 컬러/선명
        const gray = 100 - restoration;
        const blur = (100 - restoration) / 10; // 최대 10px 블러
        const bright = 0.5 + (restoration / 200); // 0.5 -> 1.0

        imgEl.style.filter = `grayscale(${gray}%) blur(${blur}px) brightness(${bright})`;
        
        hpBar.style.width = restoration + "%";
        percentTxt.innerText = restoration + "%";
        msgTxt.innerText = "역사의 색을 되찾고 있습니다...";

        if (restoration >= 100) {
            msgTxt.innerText = "복원 완료! 역사를 지켜냈습니다!";
            msgTxt.style.color = "green";
            setTimeout(() => alert("🎉 미션 성공! 🎉\n안양 만안교의 복원을 완료했습니다.\n\n(다음 유물 위치가 지도에 표시됩니다)"), 500);
        }
    }

</script>
