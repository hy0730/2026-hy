<script>
    let restoration = 0; // 현재 복원도 (0 ~ 100)
    const imgEl = document.getElementById('artifact-img');
    const barEl = document.getElementById('hp-bar');
    const txtEl = document.getElementById('percent-txt');
    const msgEl = document.getElementById('msg-box');
    const effectLayer = document.getElementById('effect-layer');

    // === 핵심 메커니즘: 복원 행동 ===
    function restore(amount, type) {
        if (restoration >= 100) return; // 이미 완료됨

        // 1. 수치 증가
        restoration += amount;
        if (restoration > 100) restoration = 100;

        // 2. 시각적 피드백 (점점 선명해짐)
        updateVisuals();

        // 3. 효과 메시지
        let msg = "";
        if(type === 'brush') msg = "조심스럽게 먼지를 털어냅니다.";
        if(type === 'hammer') msg = "돌의 균형을 맞춥니다!";
        if(type === 'glue') msg = "틈새를 메꿔 견고하게 만듭니다.";
        msgEl.innerText = msg;

        // 4. 파티클 효과 생성
        createSparkle();

        // 5. 완료 체크
        if (restoration >= 100) {
            msgEl.innerText = "🎉 복원 성공! 역사가 되살아났습니다!";
            msgEl.style.color = "green";
            msgEl.style.fontWeight = "bold";
            alert("✨ [만안교] 복원 완료! ✨\n\n정조 대왕의 효심이 깃든 다리가 본래의 모습을 되찾았습니다.");
        }
    }

    // === 비주얼 업데이트 (흑백 -> 컬러 변환 로직) ===
    function updateVisuals() {
        // 복원율 0% -> grayscale(100%), blur(5px)
        // 복원율 100% -> grayscale(0%), blur(0px)
        
        const grayVal = 100 - restoration; // 100 -> 0
        const blurVal = (100 - restoration) / 20; // 5 -> 0
        const brightVal = 0.6 + (restoration / 100 * 0.4); // 0.6 -> 1.0

        imgEl.style.filter = `grayscale(${grayVal}%) blur(${blurVal}px) brightness(${brightVal})`;
        
        // HP바 업데이트
        barEl.style.width = restoration + "%";
        txtEl.innerText = `복원율 ${restoration}%`;
    }

    // === 이펙트 (반짝임) ===
    function createSparkle() {
        const sparkle = document.createElement('div');
        sparkle.className = 'sparkle';
        // 랜덤 위치
        const x = 50 + Math.random() * 200;
        const y = 50 + Math.random() * 200;
        sparkle.style.left = x + 'px';
        sparkle.style.top = y + 'px';
        effectLayer.appendChild(sparkle);

        // 0.5초 뒤 삭제
        setTimeout(() => sparkle.remove(), 500);
    }

    // === 이미지 업로드 기능 ===
    document.getElementById('imgInput').addEventListener('change', function(e) {
        const file = e.target.files[0];
        if(file) {
            const reader = new FileReader();
            reader.onload = function(evt) {
                imgEl.src = evt.target.result;
                // 초기화
                restoration = 0;
                updateVisuals();
                msgEl.innerText = "새로운 유물 복원 시작!";
                msgEl.style.color = "#7f8c8d";
            }
            reader.readAsDataURL(file);
        }
    });

</script>
